41impl System for CollisionSystem {
42    fn name(&self) -> Cow<'static, str> {
43        "sdf_populate_world_system".into()
44    }
45
46    fn run(
47        &mut self,
48        MapContext {
49            world,
50            view_state,
51            renderer: Renderer { device, queue, .. },
52            ..
53        }: &mut MapContext,
54    ) -> SystemResult {
55        let Some((Initialized(tile_view_pattern), Initialized(symbol_buffer_pool))) =
56            world.resources.query_mut::<(
57                &mut Eventually<WgpuTileViewPattern>,
58                &mut Eventually<SymbolBufferPool>,
59            )>()
60        else {
61            return Err(SystemError::Dependencies);
62        };
63
64        if !view_state.did_camera_change() {
65            // TODO
66            // return Ok(());
67        }
68
69        let mut collision_index = CollisionIndex::new(view_state, MapMode::Continuous);
70
71        for view_tile in tile_view_pattern.iter() {
72            let coords = view_tile.coords();
73            if let Some(component) = world.tiles.query::<&SymbolLayersDataComponent>(coords) {
74                for layer in &component.layers {
75                    let mut feature_metadata = vec![
76                        SDFShaderFeatureMetadata::default();
77                        layer
78                            .features
79                            .last()
80                            .map(|feature| feature.indices.end)
81                            .unwrap_or_default()
82                    ];
83
84                    for feature in &layer.features {
85                        // calculate where tile is
86
87                        let transform = coords.transform_for_zoom(view_state.zoom());
88
89                        let pos_matrix = view_state
90                            .view_projection()
91                            .to_model_view_projection(transform);
92
93                        let zoom_factor = view_state.zoom().scale_to_tile(&coords);
94
95                        let font_scale = 6.0;
96                        let scaling = Matrix3::from_cols(
97                            Vector3::new(zoom_factor * font_scale, 0.0, 0.0),
98                            Vector3::new(0.0, zoom_factor * font_scale, 0.0),
99                            Vector3::new(0.0, 0.0, 1.0),
100                        );
101
102                        let vec3 = Vector3::new(
103                            feature.bbox.max.x as f64,
104                            feature.bbox.max.y as f64,
105                            0.0f64,
106                        );
107                        let text_anchor = Vector3::new(
108                            feature.text_anchor.x as f64,
109                            feature.text_anchor.y as f64,
110                            0.0f64,
111                        );
112
113                        let shader = pos_matrix.get()
114                            * (scaling * (vec3 - text_anchor) + text_anchor).extend(1.0);
115                        let window = view_state.clip_to_window(&shader);
116
117                        //println!("{:?}", window);
118
119                        let anchor_point =
120                            Point2D::new(feature.bbox.min.x as f64, feature.bbox.min.y as f64); // TODO
121
122                        let boxes = vec![CollisionBox {
123                            anchor: anchor_point,
124                            x1: 0.0 * (EXTENT / TILE_SIZE),
125                            y1: 0. * (EXTENT / TILE_SIZE),
126                            x2: (feature.bbox.max.x - feature.bbox.min.x) as f64, //* (EXTENT / TILE_SIZE),
127                            y2: (feature.bbox.max.y - feature.bbox.min.y) as f64, // * (EXTENT / TILE_SIZE),
128                            signed_distance_from_anchor: 0.0,
129                        }]; // TODO
130
131                        let mut projected_boxes = vec![];
132                        let collision_feature = CollisionFeature {
133                            boxes,
134                            indexed_feature: IndexedSubfeature {
135                                ref_: RefIndexedSubfeature {
136                                    index: 0,
137                                    sort_index: 0,
138                                    source_layer_name: "".to_string(),
139                                    bucket_leader_id: "".to_string(),
140                                    bucket_instance_id: 0,
141                                    collision_group_id: 0,
142                                },
143                                source_layer_name_copy: "".to_string(),
144                                bucket_leader_idcopy: "".to_string(),
145                            },
146                            along_line: false, // false if point, else true
147                        };
148                        let (placed_text, is_offscreen) = collision_index.place_feature(
149                            &collision_feature,
150                            Point2D::zero(), // shift
151                            &pos_matrix,
152                            &pos_matrix.get(), // TODO
153                            //TILE_SIZE / EXTENT,
154                            1.0,
155                            &PlacedSymbol {
156                                anchor_point,
157                                segment: 0,
158                                lower_size: 0.0,
159                                upper_size: 0.0,
160                                line_offset: [0., 0.],
161                                writing_modes: Default::default(),
162                                line: GeometryCoordinates(vec![anchor_point.cast()]), // TODO can be linestring or just a single point
163                                tile_distances: vec![],                               // TODO
164                                glyph_offsets: vec![0., 0.],                          // TODO
165                                hidden: false,
166                                vertex_start_index: 0,
167                                cross_tile_id: 0,
168                                placed_orientation: None,
169                                angle: 0.0,
170
171                                placed_icon_index: None,
172                            },
173                            view_state.zoom().scale_to_zoom_level(coords.z),
174                            6.0,
175                            false,
176                            false,
177                            false,
178                            None,                               // avoidEdges
179                            Some(|f: &IndexedSubfeature| true), // collisionGroupPredicate
180                            &mut projected_boxes,               // output
181                        );
182                        if feature.str.starts_with("Ette") {
183                            //println!("{}", feature.str);
184                            //println!("{:?}", &collision_feature.boxes);
185                            //println!("proj {:?}", &projected_boxes.get(0));
186                        }
187
188                        if placed_text {
189                            collision_index.insert_feature(
190                                collision_feature,
191                                &projected_boxes,
192                                false,
193                                55,
194                                66,
195                            );
196
197                            for index in feature.indices.clone() {
198                                let index = layer.buffer.buffer.indices[index] as usize;
199                                feature_metadata[index].opacity = 1.0;
200                            }
201                        } else {
202                            for index in feature.indices.clone() {
203                                let index = layer.buffer.buffer.indices[index] as usize;
204                                feature_metadata[index].opacity = 0.0;
205                            }
206
207                            //feature_metadata.extend(iter::repeat(SDFShaderFeatureMetadata { opacity: 0.0 }).take(feature.indices.len()))
208                        }
209                    }
210
211                    if let Some(layer_at_coords) = symbol_buffer_pool.index().get_layers(coords) {
212                        for entry in layer_at_coords {
213                            debug_assert_eq!(entry.coords, coords);
214
215                            let source_layer = entry.style_layer.source_layer.as_ref().unwrap();
216
217                            if source_layer != &layer.source_layer {
218                                continue;
219                            }
220
221                            symbol_buffer_pool.update_feature_metadata(
222                                queue,
223                                entry,
224                                &feature_metadata,
225                            );
226                        }
227                    }
228                }
229            }
230        }
231        Ok(())
232    }
