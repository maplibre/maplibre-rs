28/// maplibre/maplibre-native#4add9ea original name: getGlyphRange
29pub fn get_glyph_range(glyph: GlyphID) -> GlyphRange {
30    let mut start: u16 = (glyph / 256) * 256;
31    let mut end = start + 255;
32    if start > 65280 {
33        start = 65280;
34    }
35    if end > 65535 {
36        end = 65535;
37    }
38    start..end
