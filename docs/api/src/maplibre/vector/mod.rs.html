77impl HasTile for VectorTilesDone {
78    fn has_tile(&self, coords: WorldTileCoords, world: &World) -> bool {
79        let Some(vector_layers_indices) = world.tiles.query::<&VectorLayerBucketComponent>(coords)
80        else {
81            return false;
82        };
83
84        vector_layers_indices.done
85    }
86}

88impl<E: Environment, T: VectorTransferables> Plugin<E> for VectorPlugin<T> {
89    fn build(
90        &self,
91        schedule: &mut Schedule,
92        kernel: Rc<Kernel<E>>,
93        world: &mut World,
94        _graph: &mut RenderGraph,
95    ) {
96        let resources = &mut world.resources;
97
98        resources.insert(Eventually::<VectorBufferPool>::Uninitialized);
99        resources.insert(Eventually::<VectorPipeline>::Uninitialized);
100
101        resources
102            .get_or_init_mut::<ViewTileSources>()
103            .add_resource_query::<&Eventually<VectorBufferPool>>()
104            .add::<VectorTilesDone>();
105
106        schedule.add_system_to_stage(
107            RenderStageLabel::Extract,
108            SystemContainer::new(RequestSystem::<E, T>::new(&kernel)),
109        );
110        schedule.add_system_to_stage(
111            RenderStageLabel::Extract,
112            SystemContainer::new(PopulateWorldSystem::<E, T>::new(&kernel)),
113        );
114
115        schedule.add_system_to_stage(RenderStageLabel::Prepare, resource_system);
116        schedule.add_system_to_stage(RenderStageLabel::Queue, upload_system); // FIXME tcs: Upload updates the TileView in tileviewpattern -> upload most run before prepare
117        schedule.add_system_to_stage(RenderStageLabel::Queue, queue_system);
118    }
119}
