40/// A request for a tile at the given coordinates and in the given layers.
41pub struct VectorTileRequest {
42    pub coords: WorldTileCoords,
43    pub layers: HashSet<StyleLayer>,
44}
45
46pub fn process_vector_tile<T: VectorTransferables, C: Context>(
47    data: &[u8],
48    tile_request: VectorTileRequest,
49    context: &mut ProcessVectorContext<T, C>,
50) -> Result<(), ProcessVectorError> {
51    let mut tile = geozero::mvt::Tile::decode(data)
52        .map_err(|e| ProcessVectorError::Decoding(e.to_string().into()))?;
53
54    // Report available layers
55    let coords = &tile_request.coords;
56
57    for style_layer in &tile_request.layers {
58        let id = &style_layer.id;
59        if let (Some(paint), Some(source_layer)) = (&style_layer.paint, &style_layer.source_layer) {
60            if let Some(layer) = tile
61                .layers
62                .iter_mut()
63                .find(|layer| &layer.name == source_layer)
64            {
65                let original_layer = layer.clone();
66
67                match paint {
68                    LayerPaint::Line(_) | LayerPaint::Fill(_) => {
69                        let mut tessellator = ZeroTessellator::<IndexDataType>::default();
70
71                        if let Err(e) = layer.process(&mut tessellator) {
72                            context.layer_missing(coords, &source_layer)?;
73
74                            tracing::error!("tesselation for layer source {source_layer} at {coords} failed {e:?}");
75                        } else {
76                            context.layer_tesselation_finished(
77                                coords,
78                                tessellator.buffer.into(),
79                                tessellator.feature_indices,
80                                original_layer,
81                            )?;
82                        }
83                    }
84                    LayerPaint::Symbol(symbol_paint) => {
85                        let mut tessellator = TextTessellator::<IndexDataType>::default();
86                        let text_field = symbol_paint
87                            .text_field
88                            .clone()
89                            .unwrap_or_else(|| "name".to_string());
90                        let mut tessellator_new = TextTessellatorNew::new(text_field);
91
92                        if let Err(e) = layer.process(&mut tessellator_new) {
93                            context.layer_missing(coords, &source_layer)?;
94
95                            tracing::error!("tesselation for layer source {source_layer} at {coords} failed {e:?}");
96                        } else {
97                            tessellator_new.finish();
98                            context.symbol_layer_tesselation_finished(
99                                coords,
100                                tessellator.quad_buffer.into(),
101                                tessellator_new.quad_buffer.into(),
102                                tessellator_new.features,
103                                original_layer,
104                            )?;
105                        }
106                    }
107                    _ => {
108                        log::warn!("unhandled style layer type in {id}");
109                    }
110                }
111            } else {
112                log::warn!("layer source {source_layer} not found in vector tile");
113            }
114        } else {
115            log::error!("vector style layer {id} misses a required attribute");
116        }
117    }
118
119    // Report missing layers
120    let coords = &tile_request.coords;
121    let available_layers: HashSet<_> = tile
122        .layers
123        .iter()
124        .map(|layer| layer.name.clone())
125        .collect::<HashSet<_>>();
126
127    for layer in tile_request.layers {
128        if let Some(source_layer) = layer.source_layer {
129            if !available_layers.contains(&source_layer) {
130                context.layer_missing(coords, &source_layer)?;
131                tracing::info!(
132                    "requested source layer {source_layer} at {coords} not found in tile"
133                );
134            }
135        }
136    }
137
138    // Report index for layer
139    let mut index = IndexProcessor::new();
140
141    for layer in &mut tile.layers {
142        layer.process(&mut index).unwrap();
143    }
144
145    context.layer_indexing_finished(&tile_request.coords, index.get_geometries())?;
146
147    // Report end
148    tracing::info!("tile tessellated at {coords} finished");
149    context.tile_finished(coords)?;
150
151    Ok(())
152}

154pub struct ProcessVectorContext<T: VectorTransferables, C: Context> {
155    context: C,
156    phantom_t: PhantomData<T>,
157}
158
159impl<T: VectorTransferables, C: Context> ProcessVectorContext<T, C> {
160    pub fn new(context: C) -> Self {
161        Self {
162            context,
163            phantom_t: Default::default(),
164        }
165    }
166}
167
168impl<T: VectorTransferables, C: Context> ProcessVectorContext<T, C> {
169    pub fn take_context(self) -> C {
170        self.context
171    }
172
173    fn tile_finished(&mut self, coords: &WorldTileCoords) -> Result<(), ProcessVectorError> {
174        self.context
175            .send_back(T::TileTessellated::build_from(*coords))
176            .map_err(|e| ProcessVectorError::SendError(e))
177    }
178
179    fn layer_missing(
180        &mut self,
181        coords: &WorldTileCoords,
182        layer_name: &str,
183    ) -> Result<(), ProcessVectorError> {
184        self.context
185            .send_back(T::LayerMissing::build_from(*coords, layer_name.to_owned()))
186            .map_err(|e| ProcessVectorError::SendError(e))
187    }
188
189    fn layer_tesselation_finished(
190        &mut self,
191        coords: &WorldTileCoords,
192        buffer: OverAlignedVertexBuffer<ShaderVertex, IndexDataType>,
193        feature_indices: Vec<u32>,
194        layer_data: tile::Layer,
195    ) -> Result<(), ProcessVectorError> {
196        self.context
197            .send_back(T::LayerTessellated::build_from(
198                *coords,
199                buffer,
200                feature_indices,
201                layer_data,
202            ))
203            .map_err(|e| ProcessVectorError::SendError(e))
204    }
205
206    fn symbol_layer_tesselation_finished(
207        &mut self,
208        coords: &WorldTileCoords,
209        buffer: OverAlignedVertexBuffer<ShaderSymbolVertex, IndexDataType>,
210        new_buffer: OverAlignedVertexBuffer<ShaderSymbolVertexNew, IndexDataType>,
211        features: Vec<Feature>,
212        layer_data: tile::Layer,
213    ) -> Result<(), ProcessVectorError> {
214        self.context
215            .send_back(T::SymbolLayerTessellated::build_from(
216                *coords, buffer, new_buffer, features, layer_data,
217            ))
218            .map_err(|e| ProcessVectorError::SendError(e))
219    }
220
221    fn layer_indexing_finished(
222        &mut self,
223        coords: &WorldTileCoords,
224        geometries: Vec<IndexedGeometry<f64>>,
225    ) -> Result<(), ProcessVectorError> {
226        self.context
227            .send_back(T::LayerIndexed::build_from(
228                *coords,
229                TileIndex::Linear { list: geometries },
230            ))
231            .map_err(|e| ProcessVectorError::SendError(e))
232    }
233}
